044 //为链表排序,排序说明：第一次前两个数比较，第二次第三个数和前两个数比较，以此类推
045 int i,j;
046 int temp;
047 for(int i=1;i<stuList.length;i++){
048 temp=stuList.student[i];
049 for(j=i-1;j>=;j–){
